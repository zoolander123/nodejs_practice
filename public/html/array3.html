Программа запрашивает ввести что угодно. Если вводишь число - записывает в один массив, если слово(строку) - в другой.
 При нажатии Cancel ввод прекращается и на экран выводиться количество введенных чисел и слов.

 var numbers = [];
 var strings = [];

 do {
    var inp = prompt("Enter the number:", "");
    if ( !(isNaN(inp)) ) {
        numbers.push(inp)
    }

    else if ( isNaN(inp) ) {
        strings.push(inp)
    }

 } while(inp != null)

 alert(numbers);
 alert(strings);
