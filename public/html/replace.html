Заменить в строке в словах которые состоят больше чем из 5 букв на прописную заглавную букву

     var text = "Integer imperdiet 1 vitae odio a ullamcorper. Pellentesque 45 arcu massa, pretium vel leo vel";
     var sTxt = text.split(" ");
     var result = "";

     for(var i=0; i < sTxt.length; i++) {
        var word = sTxt[i];
        var noPunctWord = word.replace(".","").replace(",","");

        if(noPunctWord.length > 5){
            word = sTxt[i].charAt(0).toUpperCase()+sTxt[i].substr(1);
        }
        result += word+" ";
     }
     alert(result);
