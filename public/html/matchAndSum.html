text = "Morbi est metus, tristique nec 3 faucibus vel, sollicitudin a nunc. 44 Cras dictum sem eget ligula lobortis 3";

var numbers = text.match(/\d+/g);                           //матч ищет совпадения??

sum = function(m) {                                         // функция на суммирование всех цифровых значений массива
for(var s = 0, k = m.length; k; s += +m[--k]);
return s;
};

alert("Все числа в строке: " + numbers + "\nИх сумма: " + sum(numbers));
