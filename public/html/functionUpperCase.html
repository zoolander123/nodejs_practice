var str = prompt("Enter string","");

alert(firstLetter(str));

function firstLetter(str){
    var strUp = str.charAt(0).toUpperCase();  // нулевой индекс строки приводим в аппер кейс и записываем в новую переменную
    for (var i=1; i < str.length; i++){       // перебираем все индексы строки начиная с первого
        strUp += str.charAt(i);               // c каждой итерацией прибавляем индекс к созданной переменной куда поместили 0 индекс с заглавной буквой
    }

return strUp;                                 // возращаем это переменную по окончании перебора цикла
}
