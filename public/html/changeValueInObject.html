В объекте есть свойство className, которое содержит список «классов» - слов, разделенных пробелом:
var obj = { className: 'open menu' }

Создайте функцию addClass(obj, cls), которая добавляет в список класс cls, но только если его там еще нет:
addClass(obj, 'new'); // obj.className='open menu new'
addClass(obj, 'open');  // без изменений (класс уже существует)
addClass(obj, 'me'); // obj.className='open menu new me'
alert(obj.className);  // "open menu new me"
P.S. Ваша функция не должна добавлять лишних пробелов.



        var addProp = prompt("new","");
        var test = { className: 'open menu' };

            function addClass(obj, cls) {

                var classNameArr = obj.className.split(" ");    // разбиваем свойство объекта в массив

                for(var i=0; i < classNameArr.length; i++) {
                    if (classNameArr[i] == cls) return;         // идет перебор всех индексов массива
                }                                               // если свойство совпадает то выход из ФУНКЦИИ с помощью ретерн

                classNameArr.push(cls);                         // вносим в конец массива новое свойство

                var result = classNameArr.join(" ");            // собираем массив в строку. МОЖНО СОЕДЕНИТЬ СО СЛЕДУЮЩЕЙ СТРОКОЙ var result НЕ НУЖЕН

                obj.className = result;                         // присваиваем изначальному свойству объекта полученную строку

            }

        addClass(test, addProp);

        alert(test["className"]);
